Q1b

  • Use .rank() to find the top 5 rich counties in NY for each year.
    • Do not use apply().
q1b = (
   df_ny.assign( 
          ranking = df_ny.groupby('year')['pincp']
                         .rank(method = 'dense', ascending = False) 
                         )
)

q1b = (
  q1b.query('ranking <= 5')
  .sort_values(by = ['year', 'ranking'])
)
